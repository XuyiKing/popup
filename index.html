        $(".confirm").off("click").on("click", function () {
            Popup.confirm({
                title: "确认操作",
                content: "确定操作？"
            });
        });

        $(".content").off("click").on("click", function () {
            Popup.contentPopup({
                el: "#example",
                title: "内容弹窗"
            });
        })
